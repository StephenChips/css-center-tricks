Horizontal
Be applicable to fixed width and fixed height <div>.
I'm horizontal centered!
                            
<div class="container" id="hc-flexbox">
    <div class="centered">I'm horizontal centered!</div>
</div>
                            
                        
                            
/* horizontal center by flexbox */
.container {
    display: flex;
    justify-content: center;
    align-items: flex-start;
}
I'm horizontal centered
                            
<div class="container" id="hc-flexbox">
    <div class="centered">I'm horizontal centered!</div>
</div>
                            
                        
                            
/* Horizontal center by setting horizontal margins to "auto". */
.centered {
    display: block; /* Presquisite #1: must be a block-level element */
    width: 300px;   /* Presquisite #2: must have a fixed width. */
    margin-left: auto;
    margin-right: auto;
}
I'm horizontal centered
                            
<div class="container" id="hc-flexbox">
    <div class="centered">I'm horizontal centered!</div>
</div>
                            
                        
                            
/* Horizontal center by setting absoluted position */
.container {
    display: relative;
}

.centered {
    left: 0;
    right: 0;
    top: 0;

    margin-left: auto;
    margin-right: auto;
    
    /* This method can only create fixed-width horizontal centered element */
    width: 300px;
}
I'm horizontal centered
                            
<div class="container" id="hc-flexbox">
    <div class="centered">I'm horizontal centered!</div>
</div>
                            
                        
                            
/* horizontal center by setting text-align: center */
.container {
    text-align: center;
}

.centered {
    display: inline-block;
}
